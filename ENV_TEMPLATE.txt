# Создайте файл .env в корне проекта и добавьте эти строки:

TELEGRAM_BOT_TOKEN=ваш_токен_бот_от_BotFather
TELEGRAM_CHAT_ID=ваш_id_чата

# Как получить TELEGRAM_BOT_TOKEN:
# 1. Откройте Telegram и найдите @BotFather
# 2. Отправьте /newbot
# 3. Следуйте инструкциям
# 4. Скопируйте токен бота

# Как получить TELEGRAM_CHAT_ID:
# 1. Откройте Telegram и найдите @userinfobot
# 2. Отправьте /start
# 3. Бот пришлет ваш ID (например: 123456789)
# 4. Или создайте группу и добавьте туда бота - используйте ID группы

# Пример заполненного .env файла:
# TELEGRAM_BOT_TOKEN=123456789:ABCdefGHIjklMNOpqrsTUVwxyz
# TELEGRAM_CHAT_ID=123456789
